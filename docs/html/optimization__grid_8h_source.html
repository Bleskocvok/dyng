43  void add(coords pos, unsigned index) {
44  int x = std::floor((pos.x + m_w * 0.5f) / m_2k);
45  int y = std::floor((pos.y + m_h * 0.5f) / m_2k);

49  template<typename Function>
50  void for_each_around(coords pos, Function func) {
51  int pos_x = std::floor((pos.x + m_w * 0.5f) / m_2k);
52  int pos_y = std::floor((pos.y + m_h * 0.5f) / m_2k);
53  for (int y = std::max(pos_y - 1, 0); y <= std::min(pos_y + 1, m_grid_h - 1); ++y) {
54  for (int x = std::max(pos_x - 1, 0); x <= std::min(pos_x + 1, m_grid_w - 1); ++x) {
55  for (auto& i : get(x, y)) {
56  func(i);
57  }
58  }
59  }
60  }

62  void clear() {
63  m_indices.clear();
64  }

66  void reset(float w, float h, float k) {
67  m_2k = 2.0f * k;
68  m_w = w;
69  m_h = h;
70  m_grid_w = std::ceil(w / m_2k);
71  m_grid_h = std::ceil(h / m_2k);
72  m_indices.resize(m_grid_w * m_grid_h);
73  }
74 
75 private:
76  float m_2k = 0;
77  float m_w = 0;
78  float m_h = 0;
79  int m_grid_w = 0;
80  int m_grid_h = 0;
81  std::vector<std::vector<unsigned>> m_indices;
82 
83  std::vector<unsigned>& get(int x, int y) {
84  return m_indices[y * m_grid_w + x];
